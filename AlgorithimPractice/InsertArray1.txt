using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace AlgorithimPractice
{
    class Program
    {
        static int f(int index, int constant)
        {
            return index * constant;
        }

        static void makeArray(int count, int inputValue)
        {
            List<int> result = new List<int>(){ };

            for (var i = 0; i <= count; i++)
            {
                var constantInt = 2;
                var y = f(i, constantInt);

                var lowerBounds = f(i - 1, constantInt);
                var upperBounds = f(i + 1, constantInt);

                if (y > inputValue && inputValue > lowerBounds && inputValue < upperBounds)
                {
                    result.Add(inputValue);
                }

                result.Add(y);

            }

            foreach (var item in result)
            {
                Console.WriteLine(item);
            }

            Console.ReadLine();
        }

        static void Main(string[] args)
        {
            makeArray(25, 23);
        }
    }
}
